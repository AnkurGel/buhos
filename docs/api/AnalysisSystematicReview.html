def pattern_from_s(text)
  text.split("__").inject({}){|ac,v|
    key,value=v.split("_")
    ac[key]=value.to_i
    ac
  }
end

    "#{key}_#{a[key]}"
  }.join("__")
end

def pattern_order(a)
  - (1000*a["yes"].to_i + 100*a["no"].to_i + 10*a["undecided"].to_i + 1*a["ND"].to_i)
end
