def self.get_routes(app)
  lines = []
  Dir.glob("#{app.dir_base}/**/*.rb").each do |v|
    File.open(v, "rb") do |fp|
      fp.each_line do |line|
        if line =~ /(get|post|put)\s+['"].+?['"]/
          lines.push(line)
        end
      end
    end
  end
  lines
end
