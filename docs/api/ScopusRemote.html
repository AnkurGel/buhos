def xml_abstract_by_id(id, type="doi")
  uri_abstract=@scopus.get_uri_abstract(CGI.escapeElement(id), type=type)
  xml=@scopus.connect_server(uri_abstract)
  if @scopus.error
    #$log.info(@scopus.error_msg)
    @error=@scopus.error_msg
    false
  else
    xml
  end
end
