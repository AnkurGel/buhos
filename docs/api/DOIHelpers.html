def doi_without_http(doi)
  return nil if doi.nil?
  doi.gsub(/http.+doi.org\// ,"")
end

def find_doi(text)
  if text=~/\b(10[.][0-9]{4,}(?:[.][0-9]+)*\/(?:(?!["&\'<>])\S)+)\b/
    return $1
  else
    return nil
  end
end

def url_doi(doi)
  if doi=~/http/
    doi
  else
    "http://doi.org/#{doi}"
  end
end
