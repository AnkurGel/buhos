def initialize_common(votos)
  @votos=votos
  @positivos=votos.count {|v| v[:decision]=='yes'}
  @negativos=votos.count {|v| v[:decision]=='no'}
  @id=votos[0][:id]
  @text=votos[0][:text]
  @rs_id=votos[0][:systematic_review_id]
  @tag_id=votos[0][:tag_id]
end
